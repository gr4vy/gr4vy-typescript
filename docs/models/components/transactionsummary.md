```typescript
  amount: 1299,
  status: "buyer_approval_pending",
  authorizedAmount: 1299,
  capturedAmount: 1299,
  refundedAmount: 1299,
  settledAmount: 1100,
  settled: true,

  updatedAt: new Date("2013-07-16T19:23:00.000+00:00"),
  disputed: true,
};
```
